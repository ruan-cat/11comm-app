# 状态管理迁移：Vuex 到 Pinia

## 1. 核心概念对比

| 特性       | Vuex 3.x                                 | Pinia 2.x                                |
| :--------- | :--------------------------------------- | :--------------------------------------- |
| 状态定义   | `state: { ... }`                         | `const xxx = ref/reactive(...)`          |
| 修改状态   | `mutations: { SET_XXX(state, payload) }` | 直接修改 `xxx.value = ...`               |
| 异步操作   | `actions: { async loadXXX({ commit }) }` | `async function loadXXX() { ... }`       |
| 计算属性   | `getters: { computedValue(state) }`      | `computed(() => ...)`                    |
| 模块化     | modules 嵌套                             | 多个 defineStore                         |
| TypeScript | 需要额外配置                             | 原生支持，自动类型推导                   |
| DevTools   | Vue DevTools                             | Vue DevTools (自动集成)                  |
| 持久化     | 需要第三方插件                           | `pinia-plugin-persistedstate` (内置支持) |

## 2. 从 Vuex 到 Pinia 的迁移

### 2.1 Vue2 Vuex Store 示例

```javascript
// store/index.js
import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

const store = new Vuex.Store({
  state: {
    user: null,
    taskList: [],
    loading: false
  },
  mutations: {
    SET_USER(state, user) {
      state.user = user
    },
    SET_TASK_LIST(state, tasks) {
      state.taskList = tasks
    },
    SET_LOADING(state, loading) {
      state.loading = loading
    }
  },
  actions: {
    async loadTasks({ commit }) {
      commit('SET_LOADING', true)
      try {
        const result = await GetTaskList()
        commit('SET_TASK_LIST', result.tasks)
      } finally {
        commit('SET_LOADING', false)
      }
    }
  },
  getters: {
    completedTasks: state => {
      return state.taskList.filter(task => task.status === 'completed')
    }
  }
})

export default store

// 组件中使用
import { mapState, mapGetters, mapActions } from 'vuex'

export default {
  computed: {
    ...mapState(['user', 'taskList', 'loading']),
    ...mapGetters(['completedTasks'])
  },
  methods: {
    ...mapActions(['loadTasks'])
  }
}
```

### 2.2 Vue3 Pinia Store 示例

```typescript
// stores/useTaskStore.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getTaskList } from '@/api/task'

interface Task {
  id: string
  title: string
  status: 'pending' | 'completed'
  assignee: string
}

interface User {
  id: string
  name: string
  role: string
}

export const useTaskStore = defineStore('task', () => {
  // State - 使用 ref 定义响应式状态
  const user = ref<User | null>(null)
  const taskList = ref<Task[]>([])
  const loading = ref(false)

  // Getters - 使用 computed 定义计算属性
  const completedTasks = computed(() =>
    taskList.value.filter(task => task.status === 'completed')
  )

  const taskCount = computed(() => ({
    total: taskList.value.length,
    completed: completedTasks.value.length,
    pending: taskList.value.filter(task => task.status === 'pending').length
  }))

  // Actions - 直接定义异步函数
  const loadTasks = async () => {
    loading.value = true
    try {
      const result = await getTaskList()
      taskList.value = result
    } catch (error) {
      console.error('加载任务失败:', error)
      throw error
    } finally {
      loading.value = false
    }
  }

  const addTask = (task: Omit<Task, 'id'>) => {
    const newTask: Task = {
      ...task,
      id: Date.now().toString()
    }
    taskList.value.unshift(newTask)
  }

  const updateTask = (taskId: string, updates: Partial<Task>) => {
    const index = taskList.value.findIndex(task => task.id === taskId)
    if (index !== -1) {
      taskList.value[index] = { ...taskList.value[index], ...updates }
    }
  }

  const deleteTask = (taskId: string) => {
    const index = taskList.value.findIndex(task => task.id === taskId)
    if (index !== -1) {
      taskList.value.splice(index, 1)
    }
  }

  const setCurrentUser = (newUser: User) => {
    user.value = newUser
  }

  // 返回需要暴露的 state、getters 和 actions
  return {
    // State - 使用 readonly 保护状态
    user: readonly(user),
    taskList: readonly(taskList),
    loading: readonly(loading),
    // Getters
    completedTasks,
    taskCount,
    // Actions
    loadTasks,
    addTask,
    updateTask,
    deleteTask,
    setCurrentUser
  }
}, {
  // 持久化配置
  persist: {
    key: 'task-store',
    storage: {
      getItem: uni.getStorageSync,
      setItem: uni.setStorageSync
    },
    paths: ['user'] // 只持久化用户信息，任务列表不持久化
  }
})

// 组件中使用
<script setup lang="ts">
import { storeToRefs } from 'pinia'
import { useTaskStore } from '@/stores/useTaskStore'

const taskStore = useTaskStore()

// 使用 storeToRefs 解构响应式状态和 getters
const { taskList, loading, completedTasks, taskCount } = storeToRefs(taskStore)

// Actions 可以直接解构（不需要 storeToRefs）
const { loadTasks, addTask, updateTask, deleteTask } = taskStore

// 使用示例
const handleAddTask = () => {
  addTask({
    title: '新任务',
    status: 'pending',
    assignee: 'user1'
  })
}

// 页面加载时获取数据
onMounted(() => {
  loadTasks()
})
</script>
```

## 3. Pinia Store 的两种定义方式

### 3.1 Setup Stores (推荐)

```typescript
import { defineStore } from "pinia";
import { ref, computed } from "vue";

export const useCounterStore = defineStore("counter", () => {
	// State
	const count = ref(0);
	const name = ref("Eduardo");

	// Getters
	const doubleCount = computed(() => count.value * 2);

	// Actions
	function increment() {
		count.value++;
	}

	return { count, name, doubleCount, increment };
});
```

**优点**:

- 更接近 Composition API 风格
- 更灵活，可以使用任何 Vue 组合式函数
- TypeScript 类型推导更好

### 3.2 Option Stores

```typescript
import { defineStore } from "pinia";

export const useCounterStore = defineStore("counter", {
	state: () => ({
		count: 0,
		name: "Eduardo",
	}),
	getters: {
		doubleCount: (state) => state.count * 2,
	},
	actions: {
		increment() {
			this.count++;
		},
	},
});
```

**适用场景**: 从 Vuex 迁移时，保持类似的结构

## 4. Pinia 持久化配置

### 4.1 全局持久化

```typescript
// main.ts
import { createPinia } from "pinia";
import { createPersistedState } from "pinia-plugin-persistedstate";

const pinia = createPinia();
pinia.use(
	createPersistedState({
		storage: {
			getItem: uni.getStorageSync,
			setItem: uni.setStorageSync,
		},
	}),
);

app.use(pinia);
```

### 4.2 单个 Store 持久化

```typescript
export const useUserStore = defineStore(
	"user",
	() => {
		const user = ref<User | null>(null);
		const token = ref<string>("");

		return {
			user,
			token,
		};
	},
	{
		persist: {
			key: "user-store",
			storage: {
				getItem: uni.getStorageSync,
				setItem: uni.setStorageSync,
			},
			// 只持久化特定字段
			paths: ["user", "token"],
		},
	},
);
```

### 4.3 多存储策略

```typescript
export const useSettingsStore = defineStore(
	"settings",
	() => {
		const theme = ref("light");
		const language = ref("zh-CN");
		const notifications = ref(true);

		return {
			theme,
			language,
			notifications,
		};
	},
	{
		persist: [
			{
				key: "settings-theme",
				storage: {
					getItem: uni.getStorageSync,
					setItem: uni.setStorageSync,
				},
				paths: ["theme"],
			},
			{
				key: "settings-prefs",
				storage: {
					getItem: uni.getStorageSync,
					setItem: uni.setStorageSync,
				},
				paths: ["language", "notifications"],
			},
		],
	},
);
```

## 5. 组件中使用 Store

### 5.1 访问 State

```typescript
<script setup lang="ts">
import { storeToRefs } from 'pinia'
import { useTaskStore } from '@/stores/useTaskStore'

const taskStore = useTaskStore()

// ✅ 正确：使用 storeToRefs 保持响应性
const { taskList, loading } = storeToRefs(taskStore)

// ❌ 错误：直接解构会失去响应性
const { taskList, loading } = taskStore
</script>
```

### 5.2 调用 Actions

```typescript
<script setup lang="ts">
import { useTaskStore } from '@/stores/useTaskStore'

const taskStore = useTaskStore()

// ✅ 正确：Actions 可以直接解构
const { loadTasks, addTask } = taskStore

// ✅ 正确：也可以直接调用
taskStore.loadTasks()

// 使用 Actions
onMounted(() => {
  loadTasks()
})
</script>
```

### 5.3 修改 State

```typescript
<script setup lang="ts">
import { useTaskStore } from '@/stores/useTaskStore'

const taskStore = useTaskStore()

// ❌ 错误：直接修改 readonly 状态
taskStore.taskList.push(newTask) // TypeScript 错误

// ✅ 正确：通过 Actions 修改
taskStore.addTask(newTask)

// ✅ 正确：使用 $patch 批量修改（Option Stores）
taskStore.$patch({
  count: taskStore.count + 1,
  name: 'New Name'
})
</script>
```

## 6. 跨 Store 访问

```typescript
// stores/useTaskStore.ts
import { useUserStore } from "./useUserStore";

export const useTaskStore = defineStore("task", () => {
	const userStore = useUserStore();

	const loadTasks = async () => {
		// 访问其他 store 的状态
		if (!userStore.user) {
			console.warn("用户未登录");
			return;
		}

		const result = await getTaskList({
			userId: userStore.user.id,
		});
		taskList.value = result;
	};

	return {
		loadTasks,
	};
});
```

## 7. Vuex 到 Pinia 迁移检查清单

- [ ] **移除 Vuex**
  - [ ] 卸载 Vuex 依赖

  - [ ] 移除 `Vue.use(Vuex)`

- [ ] **安装 Pinia**
  - [ ] 安装 `pinia` 和 `pinia-plugin-persistedstate`
  - [ ] 在 `main.ts` 中配置 Pinia

- [ ] **迁移 State**
  - [ ] `state: { }` → `const xxx = ref()`
  - [ ] 添加 TypeScript 类型定义

- [ ] **迁移 Mutations**
  - [ ] 移除所有 mutations
  - [ ] 在 actions 中直接修改状态

- [ ] **迁移 Actions**
  - [ ] `actions: { }` → `async function xxx() { }`
  - [ ] 移除 `{ commit, state }` 参数
  - [ ] 直接修改状态而不是调用 commit

- [ ] **迁移 Getters**
  - [ ] `getters: { }` → `computed(() => ...)`
