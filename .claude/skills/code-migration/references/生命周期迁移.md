# 生命周期钩子迁移指南

## 1. 生命周期函数对照表

| Vue2 Options API | Vue3 Composition API | 触发时机             | 常见用途               |
| :--------------- | :------------------- | :------------------- | :--------------------- |
| `beforeCreate`   | `setup()`            | 实例初始化之后       | 几乎不用               |
| `created`        | `setup()`            | 实例创建完成         | 很少使用               |
| `beforeMount`    | `onBeforeMount`      | 挂载开始之前         | 罕见                   |
| `mounted`        | `onMounted`          | 挂载完成后           | DOM 操作、数据加载     |
| `beforeUpdate`   | `onBeforeUpdate`     | 数据更新，DOM 更新前 | 调试、获取更新前的状态 |
| `updated`        | `onUpdated`          | DOM 更新完成后       | DOM 操作（需谨慎）     |
| `beforeDestroy`  | `onBeforeUnmount`    | 实例销毁之前         | 清理定时器、监听器     |
| `destroyed`      | `onUnmounted`        | 实例销毁完成         | 清理工作               |
| `activated`      | `onActivated`        | keep-alive 激活时    | 缓存组件恢复           |
| `deactivated`    | `onDeactivated`      | keep-alive 停用时    | 缓存组件停用           |
| `errorCaptured`  | `onErrorCaptured`    | 捕获子组件错误       | 错误处理               |

## 2. Vue 生命周期迁移示例

### 2.1 基础生命周期迁移

**Vue2 Options API**:

```javascript
export default {
	data() {
		return {
			timer: null,
		};
	},
	mounted() {
		console.log("组件已挂载");
		this.startTimer();
	},
	beforeDestroy() {
		if (this.timer) {
			clearInterval(this.timer);
		}
	},
	methods: {
		startTimer() {
			this.timer = setInterval(() => {
				this.updateTime();
			}, 1000);
		},
		updateTime() {
			// 更新时间逻辑
		},
	},
};
```

**Vue3 Composition API**:

```typescript
<script setup lang="ts">
import { ref, onMounted, onBeforeUnmount } from 'vue'

const timer = ref<NodeJS.Timeout | null>(null)

const startTimer = () => {
  timer.value = setInterval(() => {
    updateTime()
  }, 1000)
}

const updateTime = () => {
  // 更新时间逻辑
}

/** 组件挂载时启动定时器 */
onMounted(() => {
  console.log('组件已挂载')
  startTimer()
})

/** 组件卸载前清理定时器 */
onBeforeUnmount(() => {
  if (timer.value) {
    clearInterval(timer.value)
  }
})
</script>
```

### 2.2 数据加载场景

**Vue2**:

```javascript
export default {
	data() {
		return {
			loading: false,
			userInfo: null,
		};
	},
	async mounted() {
		await this.loadUserInfo();
	},
	methods: {
		async loadUserInfo() {
			this.loading = true;
			try {
				const result = await GetUserInfo(this, {});
				this.userInfo = result.user;
			} catch (error) {
				console.error("加载失败:", error);
			} finally {
				this.loading = false;
			}
		},
	},
};
```

**Vue3**:

```typescript
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { getUserInfo } from '@/api/user'

interface UserInfo {
  id: string
  name: string
  avatar: string

const loading = ref(false)

</script>
```

### 2.3 DOM 操作场景

**Vue2**:

```javascript
export default {
	mounted() {
		// 操作 DOM
		this.$refs.scrollView.scrollTop = 100;

		// 添加事件监听
		window.addEventListener("resize", this.handleResize);
	},
	beforeDestroy() {
		// 移除事件监听
		window.removeEventListener("resize", this.handleResize);
	},
	methods: {
		handleResize() {
			console.log("Window resized");
		},
	},
};
```

**Vue3**:

```typescript
<script setup lang="ts">
import { ref, onMounted, onBeforeUnmount } from 'vue'

const scrollView = ref<HTMLElement | null>(null)

const handleResize = () => {
  console.log('Window resized')
}

/** 组件挂载后 DOM 操作 */
onMounted(() => {
  // 操作 DOM
  if (scrollView.value) {
    scrollView.value.scrollTop = 100
  }

  // 添加事件监听
  window.addEventListener('resize', handleResize)
})

/** 组件卸载前清理 */
onBeforeUnmount(() => {
  // 移除事件监听
  window.removeEventListener('resize', handleResize)
})
</script>

<template>
  <view ref="scrollView" class="scroll-view">
    <!-- 内容 -->
  </view>
</template>
```

## 3. uni-app 页面生命周期迁移

### 3.1 页面生命周期对照表

| :------------------------- | :--------------------------------------------- | :--------------- | :------------- |
| `onLoad`                   | `onLoad` (from `@dcloudio/uni-app`)            | 页面加载         | 获取页面参数   |
| `onShow`                   | `onShow` (from `@dcloudio/uni-app`)            | 页面显示         | 刷新数据       |
| `onReady`                  | `onReady` (from `@dcloudio/uni-app`)           | 页面初次渲染完成 | DOM 操作       |

| `onUnload`                 | `onUnload` (from `@dcloudio/uni-app`)          | 页面卸载         | 清理资源       |
| `onPullDownRefresh`        | `onPullDownRefresh` (from `@dcloudio/uni-app`) | 下拉刷新         | 刷新页面数据   |
| `onReachBottom`            | `onReachBottom` (from `@dcloudio/uni-app`)     | 上拉触底         | 加载更多数据   |

### 3.2 页面事件处理迁移

**Vue2 uni-app 页面**:

```javascript
export default {
	data() {
		return {
			currentPage: 1,
			hasMore: true,
			taskList: [],
			loading: false,
		};
	},
	// 页面加载
	onLoad(options) {
		console.log("页面参数:", options);
		this.loadTasks();
	},
	// 页面显示
	onShow() {
		this.refreshData();
	},
	// 下拉刷新
	onPullDownRefresh() {
		this.refreshData().then(() => {
			uni.stopPullDownRefresh();
		});
	},
	// 上拉加载
	onReachBottom() {
		if (!this.loading && this.hasMore) {
			this.loadMore();
		}
	},
	methods: {
		async refreshData() {
			this.loading = true;
			try {
				const result = await GetTaskList(this, { page: 1 });
				this.taskList = result.tasks;
				this.currentPage = 1;
				this.hasMore = true;
			} finally {
				this.loading = false;
			}
		},
		async loadMore() {
			this.loading = true;
			try {
				const result = await GetTaskList(this, {
					page: this.currentPage + 1,
				});
				this.taskList.push(...result.tasks);
				this.currentPage++;
				this.hasMore = result.tasks.length > 0;
			} finally {
				this.loading = false;
			}
		},
		async loadTasks() {
			// 加载逻辑
		},
	},
};
```

**Vue3 uni-app 页面**:

```typescript
<script setup lang="ts">
import { ref } from 'vue'
import {
  onLoad,
  onShow,
  onPullDownRefresh,
  onReachBottom
} from '@dcloudio/uni-app'
import { getTaskList } from '@/api/task'

interface Task {
  id: string
  title: string
  status: string
}

const currentPage = ref(1)
const hasMore = ref(true)
const taskList = ref<Task[]>([])
const loading = ref(false)
const loadingMore = ref(false)

/** 刷新数据 */
async function refreshData() {
  loading.value = true
  try {
    const result = await getTaskList({ page: 1, row: 10 })
    taskList.value = result.tasks
    currentPage.value = 1
    hasMore.value = true
  } catch (error) {
    console.error('刷新失败:', error)
  } finally {
    loading.value = false
  }
}

/** 加载更多 */
async function loadMoreData() {
  if (loadingMore.value || !hasMore.value) return

  loadingMore.value = true
  try {
    const result = await getTaskList({
      page: currentPage.value + 1,
      row: 10
    })
    if (result?.tasks?.length) {
      taskList.value.push(...result.tasks)
      currentPage.value++
    } else {
      hasMore.value = false
    }
  } catch (error) {
    console.error('加载更多失败:', error)
  } finally {
    loadingMore.value = false
  }
}

/** 加载初始数据 */
async function loadTasks() {
  // 加载逻辑
  await refreshData()
}

/** 页面加载 */
onLoad((options) => {
  console.log('页面参数:', options)
  loadTasks()
})

/** 页面显示 */
onShow(() => {
  refreshData()
})

/** 下拉刷新 */
onPullDownRefresh(async () => {
  try {
    await refreshData()
  } finally {
    uni.stopPullDownRefresh()
  }
})

/** 上拉加载 */
onReachBottom(() => {
  loadMoreData()
})
</script>
```

### 4.2 Composable 中使用生命周期

```typescript
// composables/usePageVisibility.ts
import { ref, onMounted, onBeforeUnmount } from 'vue'

export function usePageVisibility() {
  const isVisible = ref(true)

  const handleVisibilityChange = () => {
    isVisible.value = document.visibilityState === 'visible'
  }

  onMounted(() => {
    document.addEventListener('visibilitychange', handleVisibilityChange)
    isVisible.value = document.visibilityState === 'visible'
  })

  onBeforeUnmount(() => {
    document.removeEventListener('visibilitychange', handleVisibilityChange)
  })

  return {
    isVisible: readonly(isVisible)
  }
}

// 组件中使用
<script setup lang="ts">
import { usePageVisibility } from '@/composables/usePageVisibility'

const { isVisible } = usePageVisibility()

watch(isVisible, (visible) => {
  console.log('页面可见性变化:', visible)
})
</script>
```
