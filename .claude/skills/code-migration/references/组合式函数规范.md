# 组合式函数 (Composables) 规范

## 1. Composables vs Mixins

### 1.1 Vue2 Mixin 模式的问题

```javascript
// mixins/taskMixin.js
export default {
  data() {
    return {
      loading: false,
      error: null
    }
  },
  methods: {
    async handleRequest(requestFn) {
      this.loading = true
      this.error = null
      try {
        return await requestFn()
      } catch (error) {
        this.error = error.message
        throw error
      } finally {
        this.loading = false
      }
    }
  }
}

// 组件中使用
import taskMixin from '@/mixins/taskMixin'

export default {
  mixins: [taskMixin],
  methods: {
    async loadTasks() {
      await this.handleRequest(async () => {
        const result = await GetTaskList(this, {})
        this.taskList = result.tasks
      })
    }
  }
}
```

**Mixins 的问题**:

1. **命名冲突**: 多个 mixin 可能有相同的属性名或方法名
2. **来源不明**: 难以追踪属性或方法来自哪个 mixin
3. **隐式依赖**: mixin 之间可能存在隐式依赖关系
4. **类型推导困难**: TypeScript 难以推导 mixin 的类型

### 1.2 Vue3 Composable 模式的优势

```typescript
// composables/useRequest.ts
import { ref } from 'vue'

export function useRequest<T>(requestFn: () => Promise<T>) {
  const loading = ref(false)
  const error = ref<string | null>(null)
  const data = ref<T | null>(null)

  const execute = async (...args: any[]) => {
    loading.value = true
    error.value = null
    try {
      const result = await requestFn(...args)
      data.value = result
      return result
    } catch (err: any) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }

  return {
    loading: readonly(loading),
    error: readonly(error),
    data: readonly(data),
    execute
  }
}

// 组件中使用
<script setup lang="ts">
import { useRequest } from '@/composables/useRequest'
import { getTaskList } from '@/api/task'

const { loading, error, data, execute: loadTasks } = useRequest(getTaskList)

onMounted(() => {
  loadTasks()
})
</script>
```

**Composables 的优势**:

1. **清晰的来源**: 显式导入，明确知道功能来自哪里
2. **无命名冲突**: 可以重命名解构的变量
3. **类型推导完美**: TypeScript 完全支持
4. **灵活组合**: 可以自由组合多个 composable

## 2. Composable 编写规范

### 2.1 命名规范

- **文件名**: 小写字母 + 短横线，如 `use-request.ts`、`use-task.ts`
- **函数名**: 以 `use` 开头的驼峰命名，如 `useRequest`、`useTask`
- **导出**: 使用具名导出（named export）

```typescript
// ✅ 正确
export function useRequest<T>() {}
export function useTask() {}

// ❌ 错误
export default function request() {}
function useRequest() {} // 未导出
```

### 2.2 参数设计

```typescript
// ✅ 推荐：使用配置对象参数
export function useRequest<T>(options: {
	immediate?: boolean;
	onSuccess?: (data: T) => void;
	onError?: (error: Error) => void;
}) {
	// ...
}

// ✅ 简单场景：单个必需参数
export function useRequest<T>(requestFn: () => Promise<T>) {
	// ...
}

// ✅ 混合：必需参数 + 可选配置
export function useRequest<T>(
	requestFn: () => Promise<T>,
	options?: {
		immediate?: boolean;
		onSuccess?: (data: T) => void;
	},
) {
	// ...
}
```

### 2.3 返回值设计

```typescript
// ✅ 推荐：返回对象（可选择性解构）
export function useRequest<T>(requestFn: () => Promise<T>) {
	const loading = ref(false);
	const error = ref<string | null>(null);
	const data = ref<T | null>(null);

	const execute = async () => {
		// ...
	};

	return {
		loading: readonly(loading),
		error: readonly(error),
		data: readonly(data),
		execute,
	};
}

// 使用时可以选择性解构
const { loading, execute } = useRequest(fetchData);

// ✅ 也可以返回数组（类似 React Hooks）
export function useCounter(initialValue: number = 0) {
	const count = ref(initialValue);
	const increment = () => count.value++;
	const decrement = () => count.value--;

	return [count, increment, decrement] as const;
}

// 使用时按位置解构
const [count, increment, decrement] = useCounter(10);
```

## 3. 常用 Composable 模式

### 3.1 请求封装 Composable

```typescript
// composables/useRequest.ts
import { ref, Ref } from "vue";

interface UseRequestOptions<T> {
	immediate?: boolean;
	onSuccess?: (data: T) => void;
	onError?: (error: Error) => void;
}

export function useRequest<T>(requestFn: (...args: any[]) => Promise<T>, options: UseRequestOptions<T> = {}) {
	const { immediate = false, onSuccess, onError } = options;

	const loading = ref(false);
	const error = ref<Error | null>(null);
	const data = ref<T | null>(null);

	const execute = async (...args: any[]) => {
		loading.value = true;
		error.value = null;

		try {
			const result = await requestFn(...args);
			data.value = result;

			onSuccess?.(result);
			return result;
		} catch (err: any) {
			error.value = err;
			onError?.(err);
			throw err;
		} finally {
			loading.value = false;
		}
	};

	if (immediate) {
		execute();
	}

	return {
		loading: readonly(loading),
		error: readonly(error),
		data: readonly(data),
		execute,
	};
}
```

### 3.2 业务逻辑 Composable

```typescript
// composables/useTask.ts
import { ref, computed } from "vue";
import { getTaskList, updateTask } from "@/api/task";

export function useTask() {
	const taskList = ref<Task[]>([]);
	const loading = ref(false);
	const error = ref<string | null>(null);

	// 计算属性
	const completedTasks = computed(() => taskList.value.filter((task) => task.status === "completed"));

	const pendingTasks = computed(() => taskList.value.filter((task) => task.status === "pending"));

	const taskCount = computed(() => ({
		total: taskList.value.length,
		completed: completedTasks.value.length,
		pending: pendingTasks.value.length,
	}));

	// 加载任务列表
	const loadTasks = async () => {
		loading.value = true;
		error.value = null;
		try {
			const result = await getTaskList();
			taskList.value = result;
		} catch (err: any) {
			error.value = err.message;
			throw err;
		} finally {
			loading.value = false;
		}
	};

	// 更新任务状态
	const updateTaskStatus = async (taskId: string, status: string) => {
		const task = taskList.value.find((t) => t.id === taskId);
		if (task) {
			const originalStatus = task.status;
			task.status = status; // 乐观更新

			try {
				await updateTask({ id: taskId, status });
			} catch (err) {
				task.status = originalStatus; // 回滚
				throw err;
			}
		}
	};

	// 删除任务
	const deleteTask = (taskId: string) => {
		const index = taskList.value.findIndex((t) => t.id === taskId);
		if (index !== -1) {
			taskList.value.splice(index, 1);
		}
	};

	return {
		// 状态
		taskList: readonly(taskList),
		loading: readonly(loading),
		error: readonly(error),
		// 计算属性
		completedTasks,
		pendingTasks,
		taskCount,
		// 方法
		loadTasks,
		updateTaskStatus,
		deleteTask,
	};
}
```

### 3.3 副作用管理 Composable

```typescript
// composables/useTimer.ts
import { ref, onBeforeUnmount } from 'vue'

export function useTimer(callback: () => void, interval: number = 1000) {
  const timer = ref<NodeJS.Timeout | null>(null)
  const isRunning = ref(false)

  const start = () => {
    if (!isRunning.value) {
      timer.value = setInterval(callback, interval)
      isRunning.value = true
    }
  }

  const stop = () => {
    if (timer.value) {
      clearInterval(timer.value)
      timer.value = null
      isRunning.value = false
    }
  }

  const restart = () => {
    stop()
    start()
  }

  // 组件卸载时自动清理
  onBeforeUnmount(() => {
    stop()
  })

  return {
    isRunning: readonly(isRunning),
    start,
    stop,
    restart
  }
}

// 使用示例
<script setup lang="ts">
import { useTimer } from '@/composables/useTimer'

const updateTime = () => {
  console.log('Timer tick:', new Date())
}

const { isRunning, start, stop } = useTimer(updateTime, 1000)

onMounted(() => {
  start()
})
</script>
```

### 3.4 表单处理 Composable

```typescript
// composables/useForm.ts
import { reactive, ref } from 'vue'

interface UseFormOptions<T> {
  initialValues: T
  onSubmit: (values: T) => Promise<void> | void
  validate?: (values: T) => Record<string, string> | null
}

export function useForm<T extends Record<string, any>>(options: UseFormOptions<T>) {
  const { initialValues, onSubmit, validate } = options

  const values = reactive<T>({ ...initialValues })
  const errors = ref<Record<string, string>>({})
  const isSubmitting = ref(false)
  const isValid = ref(true)

  const setFieldValue = (field: keyof T, value: any) => {
    values[field] = value
    // 清除该字段的错误
    if (errors.value[field as string]) {
      delete errors.value[field as string]
    }
  }

  const resetForm = () => {
    Object.assign(values, initialValues)
    errors.value = {}
    isValid.value = true
  }

  const handleSubmit = async () => {
    // 验证
    if (validate) {
      const validationErrors = validate(values)
      if (validationErrors) {
        errors.value = validationErrors
        isValid.value = false
        return
      }
    }

    // 提交
    isSubmitting.value = true
    isValid.value = true
    try {
      await onSubmit(values)
    } catch (err: any) {
      errors.value = { _form: err.message }
      isValid.value = false
      throw err
    } finally {
      isSubmitting.value = false
    }
  }

  return {
    values,
    errors: readonly(errors),
    isSubmitting: readonly(isSubmitting),
    isValid: readonly(isValid),
    setFieldValue,
    resetForm,
    handleSubmit
  }
}

// 使用示例
<script setup lang="ts">
import { useForm } from '@/composables/useForm'
import { useGlobalToast } from "@/hooks/useGlobalToast";

const toast = useGlobalToast();

interface LoginForm {
  username: string
  password: string
}

const { values, errors, isSubmitting, handleSubmit, setFieldValue } = useForm<LoginForm>({
  initialValues: {
    username: '',
    password: ''
  },
  validate: (values) => {
    const errors: Record<string, string> = {}
    if (!values.username) errors.username = '用户名不能为空'
    if (!values.password) errors.password = '密码不能为空'
    return Object.keys(errors).length > 0 ? errors : null
  },
  onSubmit: async (values) => {
    await login(values)
    toast.success('登录成功')
  }
})
</script>
```

## 4. Composable 组合模式

### 4.1 Composable 调用其他 Composable

```typescript
// composables/useTaskWithPagination.ts
import { useRequest } from "./useRequest";
import { usePagination } from "./usePagination";
import { getTaskList } from "@/api/task";

export function useTaskWithPagination() {
	const { page, pageSize, total, setTotal, nextPage, prevPage, reset } = usePagination();

	const {
		loading,
		data,
		execute: fetchTasks,
	} = useRequest(
		async () => {
			const result = await getTaskList({
				page: page.value,
				pageSize: pageSize.value,
			});
			setTotal(result.total);
			return result.list;
		},
		{ immediate: true },
	);

	const loadMore = async () => {
		nextPage();
		await fetchTasks();
	};

	return {
		tasks: data,
		loading,
		page: readonly(page),
		total: readonly(total),
		loadMore,
		reset,
	};
}
```

### 4.2 条件性 Composable

```typescript
// composables/useFeature.ts
export function useFeature(enabled: Ref<boolean> | boolean) {
	const isEnabled = isRef(enabled) ? enabled : ref(enabled);

	const feature = computed(() => {
		if (!isEnabled.value) return null;
		return {
			// 功能相关的状态和方法
		};
	});

	return {
		isEnabled: readonly(isEnabled),
		feature,
	};
}
```

## 5. Composable 最佳实践

### 5.1 使用 readonly 保护状态

```typescript
// ✅ 正确：暴露 readonly 状态
export function useCounter() {
	const count = ref(0);
	const increment = () => count.value++;

	return {
		count: readonly(count), // 外部不能修改
		increment,
	};
}

// ❌ 错误：直接暴露可修改的 ref
export function useCounter() {
	const count = ref(0);
	return { count }; // 外部可以随意修改
}
```

### 5.2 提供清理函数

```typescript
export function useWebSocket(url: string) {
	const socket = ref<WebSocket | null>(null);
	const isConnected = ref(false);

	const connect = () => {
		socket.value = new WebSocket(url);
		socket.value.onopen = () => {
			isConnected.value = true;
		};
	};

	const disconnect = () => {
		socket.value?.close();
		isConnected.value = false;
	};

	// 自动清理
	onBeforeUnmount(() => {
		disconnect();
	});

	return {
		isConnected: readonly(isConnected),
		connect,
		disconnect,
	};
}
```

### 5.3 避免副作用

```typescript
// ❌ 错误：Composable 内部直接执行副作用
export function useTask() {
	const tasks = ref([]);

	// 立即执行请求（不推荐）
	getTaskList().then((result) => {
		tasks.value = result;
	});

	return { tasks };
}

// ✅ 正确：提供方法让调用方决定何时执行
export function useTask() {
	const tasks = ref([]);

	const loadTasks = async () => {
		const result = await getTaskList();
		tasks.value = result;
	};

	return {
		tasks: readonly(tasks),
		loadTasks,
	};
}

// 或者提供 immediate 选项
export function useTask(options: { immediate?: boolean } = {}) {
	const tasks = ref([]);

	const loadTasks = async () => {
		const result = await getTaskList();
		tasks.value = result;
	};

	if (options.immediate) {
		loadTasks();
	}

	return {
		tasks: readonly(tasks),
		loadTasks,
	};
}
```
